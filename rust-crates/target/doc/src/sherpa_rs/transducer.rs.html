53impl TransducerRecognizer {
54    pub fn new(config: TransducerConfig) -> Result<Self> {
55        let recognizer = unsafe {
56            let debug = config.debug.into();
57            let provider = config.provider.unwrap_or(get_default_provider());
58            let provider_ptr = cstring_from_str(&provider);
59
60            let encoder = cstring_from_str(&config.encoder);
61            let decoder = cstring_from_str(&config.decoder);
62            let joiner = cstring_from_str(&config.joiner);
63            let model_type = cstring_from_str(&config.model_type);
64            let modeling_unit = cstring_from_str(&config.modeling_unit);
65            let bpe_vocab = cstring_from_str(&config.bpe_vocab);
66            let hotwords_file = cstring_from_str(&config.hotwords_file);
67            let tokens = cstring_from_str(&config.tokens);
68            let decoding_method = cstring_from_str(&config.decoding_method);
69
70            let offline_model_config = sherpa_rs_sys::SherpaOnnxOfflineModelConfig {
71                transducer: sherpa_rs_sys::SherpaOnnxOfflineTransducerModelConfig {
72                    encoder: encoder.as_ptr(),
73                    decoder: decoder.as_ptr(),
74                    joiner: joiner.as_ptr(),
75                },
76                tokens: tokens.as_ptr(),
77                num_threads: config.num_threads,
78                debug,
79                provider: provider_ptr.as_ptr(),
80                model_type: model_type.as_ptr(),
81                modeling_unit: modeling_unit.as_ptr(),
82                bpe_vocab: bpe_vocab.as_ptr(),
83
84                // NULLs
85                telespeech_ctc: mem::zeroed::<_>(),
86                paraformer: mem::zeroed::<_>(),
87                tdnn: mem::zeroed::<_>(),
88                nemo_ctc: mem::zeroed::<_>(),
89                whisper: mem::zeroed::<_>(),
90                sense_voice: mem::zeroed::<_>(),
91                moonshine: mem::zeroed::<_>(),
92                fire_red_asr: mem::zeroed::<_>(),
93                dolphin: mem::zeroed::<_>(),
94                zipformer_ctc: mem::zeroed::<_>(),
95                canary: mem::zeroed::<_>(),
96            };
97
98            let recognizer_config = sherpa_rs_sys::SherpaOnnxOfflineRecognizerConfig {
99                model_config: offline_model_config,
100                feat_config: sherpa_rs_sys::SherpaOnnxFeatureConfig {
101                    sample_rate: config.sample_rate,
102                    feature_dim: config.feature_dim,
103                },
104                hotwords_file: hotwords_file.as_ptr(),
105                blank_penalty: config.blank_penalty,
106                decoding_method: decoding_method.as_ptr(),
107                hotwords_score: config.hotwords_score,
108
109                // NULLs
110                lm_config: mem::zeroed::<_>(),
111                rule_fsts: mem::zeroed::<_>(),
112                rule_fars: mem::zeroed::<_>(),
113                max_active_paths: mem::zeroed::<_>(),
114                hr: mem::zeroed::<_>(),
115            };
116
117            let recognizer = sherpa_rs_sys::SherpaOnnxCreateOfflineRecognizer(&recognizer_config);
118            if recognizer.is_null() {
119                bail!("SherpaOnnxCreateOfflineRecognizer failed");
120            }
121            recognizer
122        };
123
124        Ok(Self { recognizer })
125    }
126
127    pub fn transcribe(&mut self, sample_rate: u32, samples: &[f32]) -> String {
128        unsafe {
129            let stream = sherpa_rs_sys::SherpaOnnxCreateOfflineStream(self.recognizer);
130            sherpa_rs_sys::SherpaOnnxAcceptWaveformOffline(
131                stream,
132                sample_rate as i32,
133                samples.as_ptr(),
134                samples.len().try_into().unwrap(),
135            );
136            sherpa_rs_sys::SherpaOnnxDecodeOfflineStream(self.recognizer, stream);
137            let result_ptr = sherpa_rs_sys::SherpaOnnxGetOfflineStreamResult(stream);
138            let raw_result = result_ptr.read();
139            let text = cstr_to_string(raw_result.text as _);
140
141            // Free
142            sherpa_rs_sys::SherpaOnnxDestroyOfflineRecognizerResult(result_ptr);
143            sherpa_rs_sys::SherpaOnnxDestroyOfflineStream(stream);
144            text
145        }
146    }
147}
