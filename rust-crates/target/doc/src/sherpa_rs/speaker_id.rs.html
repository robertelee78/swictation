9#[derive(Debug, Default)]
10pub struct ExtractorConfig {
11    pub model: String,
12    pub provider: Option<String>,
13    pub num_threads: Option<usize>,
14    pub debug: bool,
15}
16
17#[derive(Debug)]
18pub struct EmbeddingExtractor {
19    pub(crate) extractor: *const sherpa_rs_sys::SherpaOnnxSpeakerEmbeddingExtractor,
20    pub embedding_size: usize,
21}
22
23impl EmbeddingExtractor {
24    pub fn new(config: ExtractorConfig) -> Result<Self> {
25        let provider = config.provider.unwrap_or(get_default_provider());
26
27        let num_threads = config.num_threads.unwrap_or(1);
28        let debug = config.debug.into();
29
30        let model_path = PathBuf::from(&config.model);
31        if !model_path.exists() {
32            bail!("model not found at {}", model_path.display())
33        }
34        let model = cstring_from_str(&config.model);
35        let provider = cstring_from_str(&provider);
36
37        let extractor_config = sherpa_rs_sys::SherpaOnnxSpeakerEmbeddingExtractorConfig {
38            debug,
39            model: model.as_ptr(),
40            num_threads: num_threads as i32,
41            provider: provider.as_ptr(),
42        };
43        let extractor =
44            unsafe { sherpa_rs_sys::SherpaOnnxCreateSpeakerEmbeddingExtractor(&extractor_config) };
45        // Assume embedding size is known or can be retrieved
46        let embedding_size =
47            unsafe { sherpa_rs_sys::SherpaOnnxSpeakerEmbeddingExtractorDim(extractor) }
48                .try_into()
49                .unwrap();
50        Ok(Self {
51            extractor,
52            embedding_size,
53        })
54    }
55
56    pub fn compute_speaker_embedding(
57        &mut self,
58        samples: Vec<f32>,
59        sample_rate: u32,
60    ) -> Result<Vec<f32>> {
61        unsafe {
62            let stream =
63                sherpa_rs_sys::SherpaOnnxSpeakerEmbeddingExtractorCreateStream(self.extractor);
64            if stream.is_null() {
65                bail!("Failed to create SherpaOnnxOnlineStream");
66            }
67
68            sherpa_rs_sys::SherpaOnnxOnlineStreamAcceptWaveform(
69                stream,
70                sample_rate as i32,
71                samples.as_ptr(),
72                samples.len() as i32,
73            );
74            sherpa_rs_sys::SherpaOnnxOnlineStreamInputFinished(stream);
75
76            if !self.is_ready(stream) {
77                bail!("Embedding extractor is not ready");
78            }
79
80            let embedding_ptr = sherpa_rs_sys::SherpaOnnxSpeakerEmbeddingExtractorComputeEmbedding(
81                self.extractor,
82                stream,
83            );
84            if embedding_ptr.is_null() {
85                bail!("Failed to compute speaker embedding");
86            }
87            tracing::debug!("using dimensions {}", self.embedding_size);
88            let embedding = std::slice::from_raw_parts(embedding_ptr, self.embedding_size).to_vec();
89            // Free
90            sherpa_rs_sys::SherpaOnnxDestroyOnlineStream(stream);
91            sherpa_rs_sys::SherpaOnnxSpeakerEmbeddingExtractorDestroyEmbedding(embedding_ptr);
92            Ok(embedding)
93        }
94    }
95
96    #[allow(clippy::missing_safety_doc)]
97    pub unsafe fn is_ready(
98        &mut self,
99        stream: *const sherpa_rs_sys::SherpaOnnxOnlineStream,
100    ) -> bool {
101        unsafe {
102            let result =
103                sherpa_rs_sys::SherpaOnnxSpeakerEmbeddingExtractorIsReady(self.extractor, stream);
104            result != 0
105        }
106    }
107}
